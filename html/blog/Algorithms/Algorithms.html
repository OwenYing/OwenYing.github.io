ListNode head = p;
//2,Add
ListNode tempResult = new ListNode(0);
p.next = tempResult;
p = p.next;
  • Add reversely
    • //1, Create HeadNode
ListNode newHead = new ListNode(-1);
while(head != null)
{
    ListNode temp = head.next;//tempNode
    head.next = newHead.next;
    newHead.next = head;
    head = temp;
}
